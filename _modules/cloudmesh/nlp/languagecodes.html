# Codes for the Representation of Names of Languages
# Codes arranged alphabetically by alpha-3/ISO 639-2 Code
#
# Note: ISO 639-2 is the alpha-3 code in Codes for the representation
# of names of languages-- Part 2. There are 21 languages that have alternative
# codes for bibliographic or terminology purposes. In those cases, each is listed
# separately and they are designated as "B" (bibliographic) or "T" (terminology).
# In all other cases there is only one ISO 639-2 code. Multiple codes assigned to
# the same language are to be considered synonyms. ISO 639-1 is the alpha-2 code.

# https://www.loc.gov/standards/iso639-2/php/code_list.php

# Codes last updated 2017-12-21

    def get(cls, mode="bibliographic"):
        # Note: ISO 639-2 is the alpha-3 code in Codes for the representation
        # of names of languages-- Part 2. There are 21 languages that have alternative
        # codes for bibliographic or terminology purposes. In those cases, each is listed
        # separately and they are designated as "B" (bibliographic) or "T" (terminology).
        # In all other cases there is only one ISO 639-2 code. Multiple codes assigned to
        # the same language are to be considered synonyms. ISO 639-1 is the alpha-2 code.

        # https://www.loc.gov/standards/iso639-2/php/code_list.php

        # Codes last updated 2017-12-21

        # "ISO 639-2 Code | ISO 639-1 Code | English name of Language | French name of Language | German name of Language"

        codes = []
        for code in cls.codes():

            iso_639_2, iso_639_1, english, frensh, german = [value.strip() for value in code.split()]
            if mode.lower() in ["b", "bibliogaraphic"]:
                if "(B)" in iso_639_2:
                    codes.append(code.replace("(B)").strip())
            elif mode.lower() in ["t", "terminology"]:
                if "(T)" in iso_639_2:
                    codes.append(code.replace("(T)").strip())
            codes = cls.codes

        return codes
